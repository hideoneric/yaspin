# Yaspin Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single `index.html` file that spins a randomized Yasuo skin wheel, shows a splash art reveal, and lets the user toggle which skins are in the pool — with state persisted in localStorage.

**Architecture:** One self-contained HTML file with all CSS and JS inline. The wheel is an SVG element re-drawn whenever the active skin list changes. Spin logic pre-selects the winning skin, computes the target rotation angle, then runs a CSS transition; the result card fades in after the transition ends.

**Tech Stack:** Vanilla HTML/CSS/JS, SVG, Data Dragon CDN for images, localStorage for persistence.

  ```js
  ];

  const DDR = 'https://ddragon.leagueoflegends.com/cdn/img/champion';
  const thumbUrl = num => `${DDR}/loading/Yasuo_${num}.jpg`;
  const splashUrl = num => `${DDR}/splash/Yasuo_${num}.jpg`;

  // ── STATE ──
  const LS_KEY = 'yaspin_enabled_skins';

  function loadEnabled() {
    try {
      const raw = localStorage.getItem(LS_KEY);
      if (raw) return new Set(JSON.parse(raw));
    } catch (_) {}
    return new Set(SKINS.map(s => s.num)); // default: all enabled
  }

  function saveEnabled(enabledSet) {
    localStorage.setItem(LS_KEY, JSON.stringify([...enabledSet]));
  }

  let enabled = loadEnabled();

  // ── SIDEBAR ──
  const skinList = document.getElementById('skin-list');

  function renderSidebar() {
    skinList.innerHTML = '';
    SKINS.forEach(skin => {
      const isActive = enabled.has(skin.num);
      const item = document.createElement('div');
      item.className = 'skin-item ' + (isActive ? 'active' : 'inactive');
      item.dataset.num = skin.num;
      item.innerHTML = `
        <img class="skin-thumb" src="${thumbUrl(skin.num)}" alt="${skin.name}" loading="lazy"/>
        <span class="skin-label">${skin.name}</span>
        <div class="toggle-dot"></div>
      `;
      item.addEventListener('click', () => toggleSkin(skin.num));
      skinList.appendChild(item);
    });
  }

  function toggleSkin(num) {
    if (enabled.has(num)) {
      enabled.delete(num);
    } else {
      enabled.add(num);

    saveEnabled(enabled);
    renderSidebar();
    renderWheel();
    updateSpinBtn();

  document.getElementById('toggle-all-btn').addEventListener('click', () => {
    if (enabled.size === SKINS.length) {

      SKINS.forEach(s => enabled.add(s.num));
    }
    saveEnabled(enabled);
    renderSidebar();
    renderWheel();
    updateSpinBtn();
  });

  function updateSpinBtn() {
    const spinBtn = document.getElementById('spin-btn');
    const msg = document.getElementById('no-skins-msg');
    if (enabled.size === 0) {
      spinBtn.disabled = true;
      msg.style.display = 'block';
    } else {
      spinBtn.disabled = false;
      msg.style.display = 'none';
    }
  }

  updateSpinBtn();
  ```

  ```js
  function renderWheel() {
    const svg = document.getElementById('wheel-svg');
    const cx = 180, cy = 180, r = 160, innerR = 28;
    const activeSkins = SKINS.filter(s => enabled.has(s.num));

    svg.innerHTML = '';

    if (activeSkins.length === 0) {
      // Empty state — just draw blank circle
      const circle = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
      circle.setAttribute('cx', cx); circle.setAttribute('cy', cy); circle.setAttribute('r', r);
      circle.setAttribute('fill', '#130326'); circle.setAttribute('stroke', '#7a5aaa'); circle.setAttribute('stroke-width', '2');
      svg.appendChild(circle);
      return;
    }

    const segAngle = 360 / activeSkins.length;
    const fills = ['#241050', '#130326'];

    activeSkins.forEach((skin, i) => {
      const startAngle = i * segAngle - 90; // -90 so first segment starts at top
      const endAngle = startAngle + segAngle;
      const startRad = startAngle * Math.PI / 180;
      const endRad = endAngle * Math.PI / 180;

      const x1 = cx + r * Math.cos(startRad);
      const y1 = cy + r * Math.sin(startRad);
      const x2 = cx + r * Math.cos(endRad);
      const y2 = cy + r * Math.sin(endRad);
      const largeArc = segAngle > 180 ? 1 : 0;

      // Segment path
      const path = document.createElementNS('http://www.w3.org/2000/svg', 'path');
      if (activeSkins.length === 1) {
        // Full circle for single skin
        path.setAttribute('d', `M ${cx} ${cy - r} A ${r} ${r} 0 1 1 ${cx - 0.01} ${cy - r} Z`);
      } else {
        path.setAttribute('d',
          `M ${cx} ${cy} L ${x1} ${y1} A ${r} ${r} 0 ${largeArc} 1 ${x2} ${y2} Z`
        );
      }
      path.setAttribute('fill', fills[i % 2]);
      path.setAttribute('stroke', '#7a5aaa');
      path.setAttribute('stroke-width', '1.5');
      svg.appendChild(path);

      // Label — positioned in the middle of the segment arc, ~65% out from center
      const midAngle = (startAngle + segAngle / 2) * Math.PI / 180;
      const labelR = r * 0.65;
      const lx = cx + labelR * Math.cos(midAngle);
      const ly = cy + labelR * Math.sin(midAngle);
      const rotateDeg = startAngle + segAngle / 2 + 90; // keep text readable

      const text = document.createElementNS('http://www.w3.org/2000/svg', 'text');
      text.setAttribute('x', lx);
      text.setAttribute('y', ly);
      text.setAttribute('text-anchor', 'middle');
      text.setAttribute('dominant-baseline', 'middle');
      text.setAttribute('fill', '#d8c8f0');
      text.setAttribute('font-size', activeSkins.length > 8 ? '8' : '9');
      text.setAttribute('letter-spacing', '0.5');
      text.setAttribute('transform', `rotate(${rotateDeg}, ${lx}, ${ly})`);
      // Shorten name: strip "Yasuo" to save space
      text.textContent = skin.name.replace(' Yasuo', '').replace('Yasuo', '').trim();
      svg.appendChild(text);
    });

    // Outer glow ring
    const ring = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
    ring.setAttribute('cx', cx); ring.setAttribute('cy', cy); ring.setAttribute('r', r + 3);
    ring.setAttribute('fill', 'none'); ring.setAttribute('stroke', '#7a5aaa33'); ring.setAttribute('stroke-width', '3');
    svg.appendChild(ring);

    // Center circle
    const center = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
    center.setAttribute('cx', cx); center.setAttribute('cy', cy); center.setAttribute('r', innerR);
    center.setAttribute('fill', '#08001a'); center.setAttribute('stroke', '#7a5aaa'); center.setAttribute('stroke-width', '2');
    svg.appendChild(center);

    const centerText = document.createElementNS('http://www.w3.org/2000/svg', 'text');
    centerText.setAttribute('x', cx); centerText.setAttribute('y', cy);
    centerText.setAttribute('text-anchor', 'middle'); centerText.setAttribute('dominant-baseline', 'middle');
    centerText.setAttribute('fill', '#7a5aaa'); centerText.setAttribute('font-size', '8');
    centerText.setAttribute('letter-spacing', '1');
    centerText.textContent = 'YASUO';
    svg.appendChild(centerText);
  }

  renderWheel();
  ```

  ```js
  // ── SPIN ──
  function spin() {
    if (isSpinning) return;
    const activeSkins = SKINS.filter(s => enabled.has(s.num));
    if (activeSkins.length === 0) return;

    isSpinning = true;
    document.getElementById('spin-btn').disabled = true;

    // Pick winner
    const winnerIdx = Math.floor(Math.random() * activeSkins.length);
    const winner = activeSkins[winnerIdx];

    // Calculate target rotation:
    // Each segment is 360/n degrees. Segment i starts at i*(360/n) degrees (from top, -90 offset).
    // We want the pointer (top, 270deg in SVG space) to land in the middle of winnerIdx's segment.
    const n = activeSkins.length;
    const segAngle = 360 / n;
    // Middle of winner segment in wheel-space (0 = top):
    const winnerMidAngle = winnerIdx * segAngle + segAngle / 2;
    // We need to rotate so that winnerMidAngle aligns with 0 (top/pointer).
    // Extra full spins for drama (5 full rotations minimum).
    const extraSpins = (5 + Math.floor(Math.random() * 3)) * 360;
    const targetRotation = currentRotation + extraSpins + (360 - (currentRotation % 360)) % 360 + (360 - winnerMidAngle) % 360;

    const svg = document.getElementById('wheel-svg');
    svg.style.transition = 'transform 4s cubic-bezier(0.17, 0.67, 0.12, 1.0)';
    svg.style.transform = `rotate(${targetRotation}deg)`;
    currentRotation = targetRotation;

    svg.addEventListener('transitionend', () => {
      isSpinning = false;
      showResult(winner);
    }, { once: true });
  }

  document.getElementById('spin-btn').addEventListener('click', spin);
  ```
